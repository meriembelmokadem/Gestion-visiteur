    <script>
        // Configuration de l'API
        const API_URL = 'http://localhost/gestion_visiteurs/api.php';
        let currentUserId = null;

        // Login
        document.getElementById('loginForm').addEventListener('submit', async (e) => {
            e.preventDefault();
            
            const username = document.getElementById('login_username').value;
            const password = document.getElementById('login_password').value;
            
            try {
                const response = await fetch(`${API_URL}?request=login`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({username, password})
                });
                
                const data = await response.json();
                
                if(data.user) {
                    currentUserId = data.user.id;
                    document.getElementById('currentUser').textContent = data.user.username;
                    document.getElementById('currentRole').textContent = data.user.role;
                    document.getElementById('loginScreen').style.display = 'none';
                    document.getElementById('mainApp').classList.add('active');
                    loadAllData();
                } else {
                    alert(data.message);
                }
            } catch(error) {
                alert('Erreur de connexion: ' + error);
            }
        });

        // Logout
        function logout() {
            if(confirm('Voulez-vous vraiment vous d√©connecter?')) {
                document.getElementById('loginScreen').style.display = 'block';
                document.getElementById('mainApp').classList.remove('active');
                document.getElementById('loginForm').reset();
            }
        }

        // Load all data
        async function loadAllData() {
            await loadVisiteurs();
            await loadUsers();
            await loadObservations();
            await loadStats();
            await populateUserSelect();
        }

        // Load Visiteurs
        async function loadVisiteurs() {
            try {
                const response = await fetch(`${API_URL}?request=visiteurs`);
                const visiteurs = await response.json();
                
                const tbody = document.getElementById('visiteursTable');
                tbody.innerHTML = visiteurs.map(v => `
                    <tr>
                        <td>${v.id_visiteur}</td>
                        <td>${v.nom} ${v.prenom}</td>
                        <td>${v.CIN}</td>
                        <td><span class="badge badge-${v.type_visiteur.toLowerCase()}">${v.type_visiteur}</span></td>
                        <td>${v.but_visite}</td>
                        <td>${new Date(v.date_visite).toLocaleString('fr-FR')}</td>
                        <td class="action-btns">
                            <button class="btn-delete" onclick="deleteVisiteur(${v.id_visiteur})">üóëÔ∏è</button>
                        </td>
                    </tr>
                `).join('');
            } catch(error) {
                console.error('Erreur:', error);
            }
        }

        // Load Users
        async function loadUsers() {
            try {
                const response = await fetch(`${API_URL}?request=users`);
                const users = await response.json();
                
                const tbody = document.getElementById('usersTable');
                tbody.innerHTML = users.map(u => `
                    <tr>
                        <td>${u.id_user}</td>
                        <td>${u.username}</td>
                        <td>${u.role}</td>
                        <td>${u.heure_debut} - ${u.heure_fin}</td>
                        <td class="action-btns">
                            <button class="btn-delete" onclick="deleteUser(${u.id_user})">üóëÔ∏è</button>
                        </td>
                    </tr>
                `).join('');
            } catch(error) {
                console.error('Erreur:', error);
            }
        }

        // Load Observations

            tbody.innerHTML = filtered.map(v => `
